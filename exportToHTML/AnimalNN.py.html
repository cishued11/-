# 替换原AnimalNN.py中的AnimalModel类
from torch import nn
import torch


class AnimalModel(nn.Module):
    def __init__(self, num_classes=90, device="cuda" if torch.cuda.is_available() else "cpu"):
        super().__init__()
        self.num_classes = num_classes
        self.device = device

        # 增强版卷积层：增加通道数+多一层卷积+BN/ReLU优化
        self.backbone = nn.Sequential(
            # 第1组：3→32通道（原10），更小卷积核+padding（保持尺寸）
            nn.Conv2d(3, 32, kernel_size=3, padding=1),  # 100×100→100×100
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),  # 节省内存
            nn.MaxPool2d(2, 2),  # 100→50

            # 第2组：32→64通道（原20）
            nn.Conv2d(32, 64, kernel_size=3, padding=1),  # 50→50
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),  # 50→25

            # 新增第3组：64→128通道
            nn.Conv2d(64, 128, kernel_size=3, padding=1),  # 25→25
            nn.BatchNorm2d(128),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),  # 25→12

            # 新增第4组：128→256通道
            nn.Conv2d(128, 256, kernel_size=3, padding=1),  # 12→12
            nn.BatchNorm2d(256),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),  # 12→6

            # 正则化：适度Dropout（原0.5→0.3，避免过度丢弃特征）
            nn.Dropout(p=0.3),
            nn.Flatten(),  # 256×6×6=9216（原2430）

            # 增强版全连接层：适配更多特征
            nn.Linear(9216, 1024),  # 原2430→300
            nn.ReLU(inplace=True),
            nn.Dropout(p=0.2),  # 新增轻量正则化
            nn.Linear(1024, 512),
            nn.ReLU(inplace=True),
            nn.Linear(512, num_classes)
        )
        self.to(self.device)

    def forward(self, x):
        return self.backbone(x)
